function swapCase(str) { let result = ""; for (let i = 0; i < str.length; i++) { let charCode = str.charCodeAt(i); // Check if the character is a lowercase letter (ASCII: 97-122) if (charCode >= 97 && charCode <= 122) { result += String.fromCharCode(charCode - 32); // Convert to uppercase } // Check if the character is an uppercase letter (ASCII: 65-90) else if (charCode >= 65 && charCode <= 90) { result += String.fromCharCode(charCode + 32); // Convert to lowercase } else { result += str[i]; // Keep non-alphabet characters unchanged } } return result; } // Example usage: console.log(swapCase("Hello World!")); // Output: hELLO wORLD!
